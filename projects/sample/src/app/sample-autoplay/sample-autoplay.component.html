            @Component({
                selector: 'app-player',
                standalone: true,
                imports: [ByteArkPlayerContainer],
                template: `
                    <byteark-player-container [options]="options"/>
                `
            })
            export class PlayerComponent {
                options: ByteArkPlayerContainerProps = {
                    fluid: true,
                    autoplay: 'any',
                    lazyload: true,
                    aspectRatio: '16:9',
                    poster: videos[0].poster,
                    sources: [
                        {
                            src: videos[0].src,
                            type: videos[0].type,
                            title: videos[0].title
                        }
                    ],
                    onPlayerCreated: () => {
                        console.log('Created!');
                    },
                    onReady: () => {
                        console.log('Ready!');
                    }
                }
            }
